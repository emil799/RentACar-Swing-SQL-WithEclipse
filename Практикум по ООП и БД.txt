CREATE TABLE PERSON(
ID IDENTITY PRIMARY KEY,
NAME NVARCHAR(25) NOT NULL,
TELEFON NVARCHAR(10) NOT NULL,
EGN NVARCHAR(10) NOT NULL
);

CREATE TABLE CAR(

ID IDENTITY PRIMARY KEY,
MARKA NVARCHAR(25) NOT NULL,
NOMER NVARCHAR(10) NOT NULL,
YEAR_ INT NOT NULL

);

CREATE TABLE RENTA(

ID IDENTITY PRIMARY KEY,
ID_PERSON INT NOT NULL,
ID_CAR INT NOT NULL,
DATE NVARCHAR(25),
CENA DECIMAL (6,2) NOT NULL,
FOREIGN KEY(ID_PERSON) REFERENCES PERSON(ID),
FOREIGN KEY(ID_CAR) REFERENCES CAR(ID)

);
